41  [Fact]
42  public void TestEqualsFunction()
43  {
44  Assert.True(new Real(0).Equals(new Real(0)));
45  Assert.False(new Real(100).Equals(new Real(150)));
46  Assert.True(new Real(500).Equals(new Real(500.0)));
47  Assert.True(Real.PositiveInfinity.Equals(Real.PositiveInfinity));
48  Assert.True(Real.NegativeInfinity.Equals(Real.NegativeInfinity));
49  Assert.False(Real.PositiveInfinity.Equals(Real.NegativeInfinity));
50  // Although NaN compares as unequal to itself when using the == operator, the .Equals method <i>does</i>
51  // detect NaNs as equal to themselves. Anything else would break the IEquatable/IEquatable<T> contract
52  // and also break using this as a hash table key.
53  Assert.True(Real.NaN.Equals(Real.NaN));
54  }

56  [Fact]
57  public void TestEqualsOperator()
58  {
59  Assert.True(new Real(0) == new Real(0));
60  Assert.False(new Real(100) == new Real(150));
61  Assert.True(new Real(500) == new Real(500.0));
62  Assert.True(Real.PositiveInfinity == Real.PositiveInfinity);
63  Assert.True(Real.NegativeInfinity == Real.NegativeInfinity);
64  Assert.False(Real.PositiveInfinity == Real.NegativeInfinity);
65
